CREATE TABLE AMISDBA.EHCHRCDEV
(
    EHS_PLAN_MANAGE_SN      NUMBER(22) NOT NULL,
    EHS_DTL_PLAN_MANAGE_SN  NUMBER(22) NOT NULL,
    DSCP_EVAL_QUEST_SN      NUMBER(22) NOT NULL,
    EVAL_SCOR1              NUMBER(12,2) NOT NULL,
    EVAL_SCOR2              NUMBER(12,2) NOT NULL,
    EVAL_SCOR3              NUMBER(12,2) NOT NULL,
    FRST_ENTR_EMPNO         VARCHAR2(7) NOT NULL,
    FRST_ENTR_DTM           DATE NOT NULL,
    FINL_UPID_EMPNO         VARCHAR2(7) NOT NULL,
    FINL_UPDT_DTM           DATE NOT NULL,
    DSCP_EVAL_QUEST_CLAU_CD VARCHAR2(4),
    DSCP_EVAL_QUEST_CLAU_NM VARCHAR2(100),
    DSCP_TYP_CD             VARCHAR2(3),
    DSCP_TYP_NM             VARCHAR2(60),
    DSCP_EVAL_QUEST_CNTE    VARCHAR2(1000),
    DOM_ST_SECT_SCOR        NUMBER(12,2),
    DOM_END_SECT_SCOR       NUMBER(12,2),
    SORT_SEQ                NUMBER(10)
)
TABLESPACE TS_ERP_D
INITRANS 5
STORAGE
(
    INITIAL 64K
    NEXT 1M
)
NOCOMPRESS;
